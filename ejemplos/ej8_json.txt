# JSON simplificado (LL(1))
Json -> Value

Value -> string
Value -> number
Value -> Object
Value -> Array
Value -> true
Value -> false
Value -> null

Object -> { MembersOpt }
MembersOpt -> Members | ε
Members -> Pair MembersTail
MembersTail -> , Pair MembersTail | ε
Pair -> string : Value

Array -> [ ElementsOpt ]
ElementsOpt -> Elements | ε
Elements -> Value ElementsTail
ElementsTail -> , Value ElementsTail | ε
